import{ useEffect, useState } from 'react'
import Cards from './Cards'
import axios from 'axios'

function Course() {
    const [book, setBook] = useState([])
    useEffect(()=>{
        const getBook = async()=>{
            try {
                const res = await axios.get("http://localhost:7001/book")
                console.log(res.data)
                setBook(res.data)
            } catch (error) {
               console.log(error) 
            }
        }
        getBook()
    }, [])
    return (
        <>
            <div className=' md:max-w-screen-base container dark:bg-black bg-slate-200 mx-auto md:px-24 px-2 py-4 flex flex-col'>
                <div className='mt-28 items-center justify-center text-center -pb-10'>
                    <h1 className='text-2xl  md:text-4xl'>
                        We are delighted to have you <span className='text-pink-500'>here!</span>
                    </h1>
                    <div className='mt-12'>
                        Welcome to <span className='text-pink-500'>Books</span> section, your gateway
                        to a world of literary wonders. Discover an extensive collection of books across all genres,
                        handpicked to ignite your imagination. Whether you are seeking timeless classics or the latest bestsellers,
                        we have something for every reader. Dive into your next great adventure <span className='text-pink-500'>today!</span>
                        <span className='mt-12 grid grid-cols-1 md:grid-cols-3 items-start'>
                            {book.map((item) => (
                                <Cards key={item.id} item={item} />
                            ))}
                        </span>
                    </div>
                </div>
            </div>
        </>
    )
}

export default Course
